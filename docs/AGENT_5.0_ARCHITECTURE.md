### Database Schema

```sql
-- Clients
CREATE TABLE clients (
    client_id INTEGER PRIMARY KEY,
    name TEXT,
    type TEXT, -- 'individual', 'estate', 'trust', 'business'
    contact_email TEXT,
    contact_phone TEXT,
    notes TEXT
);

-- Matters (Cases)
CREATE TABLE matters (
    matter_id INTEGER PRIMARY KEY,
    client_id INTEGER,
    caption TEXT,
    jurisdiction TEXT,
    case_number TEXT,
    case_type TEXT,
    status TEXT, -- 'active', 'settled', 'dismissed', 'won', 'lost'
    priority TEXT, -- 'high', 'medium', 'low'
    total_damages REAL,
    FOREIGN KEY (client_id) REFERENCES clients(client_id)
);

-- Tasks & Deadlines
CREATE TABLE tasks (
    task_id INTEGER PRIMARY KEY,
    matter_id INTEGER,
    description TEXT,
    due_date TEXT,
    status TEXT, -- 'pending', 'in_progress', 'completed'
    assigned_role INTEGER, -- References Agent 5.0 role number
    FOREIGN KEY (matter_id) REFERENCES matters(matter_id)
);

-- Documents
CREATE TABLE documents (
    document_id INTEGER PRIMARY KEY,
    matter_id INTEGER,
    title TEXT,
    document_type TEXT, -- 'pleading', 'evidence', 'correspondence', etc.
    file_path TEXT,
    created_date TEXT,
    tags TEXT,
    FOREIGN KEY (matter_id) REFERENCES matters(matter_id)
);

-- Calendar
CREATE TABLE calendar_events (
    event_id INTEGER PRIMARY KEY,
    matter_id INTEGER,
    event_type TEXT, -- 'hearing', 'deadline', 'reminder'
    event_date TEXT,
    description TEXT,
    location TEXT,
    FOREIGN KEY (matter_id) REFERENCES matters(matter_id)
);
```
