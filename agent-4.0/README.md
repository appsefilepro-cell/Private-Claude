#### PDF Toolkit
```python
from agent_4.tools import PDFProcessor

processor = PDFProcessor()

# Read PDF
result = processor.read_pdf('document.pdf')

# Extract text
text = processor.extract_text('document.pdf', page_numbers=[1, 2, 3])

# Extract tables
tables = processor.extract_tables('document.pdf')

# Generate PDF
processor.generate_pdf(
    output_path='report.pdf',
    content=[
        {'type': 'title', 'text': 'Financial Report'},
        {'type': 'text', 'text': 'Generated by Agent X5.0'}
    ],
    title='Monthly Report'
)

# Merge PDFs
processor.merge_pdfs(
    input_paths=['doc1.pdf', 'doc2.pdf'],
    output_path='merged.pdf'
)

# Encrypt PDF
processor.encrypt_pdf('document.pdf', 'encrypted.pdf', 'password123')
```

#### CSV Handler
```python
from agent_4.tools import CSVHandler

handler = CSVHandler()

# Parse CSV
result = handler.parse(file_path='data.csv')

# Transform data
handler.transform([
    {'type': 'convert', 'column': 'price', 'to_type': 'float'},
    {'type': 'calculate', 'name': 'total', 'formula': 'price * quantity'}
])

# Analyze data
analysis = handler.analyze()

# Filter data
handler.filter([
    {'column': 'price', 'operator': '>', 'value': 100}
])

# Export
handler.export(output_path='processed.csv', format='csv')
```

### ðŸ”„ Inter-Agent Communication

```python
from agent_4.orchestrator import AgentCoordinator

# Initialize coordinator
coordinator = AgentCoordinator()
coordinator.start()

# Create workflow
workflow = coordinator.create_workflow(
    workflow_name="Data Processing Pipeline",
    tasks=[
        {
            'type': 'parse_csv',
            'data': {'file': 'data.csv'},
            'capability': 'multi_asset'
        },
        {
            'type': 'analyze',
            'data': {'analysis_type': 'financial'},
            'capability': 'quantum_ai'
        },
        {
            'type': 'generate_report',
            'data': {'format': 'pdf'},
            'capability': 'basic_legal'
        }
    ],
    parallel=False  # Set to True for parallel execution
)

# Check system status
status = coordinator.get_system_status()
print(f"Active agents: {status['total_agents']}")
print(f"Active tasks: {status['active_tasks']}")
```

## Quick Start

### Installation

```bash
# Clone repository
git clone https://github.com/appsefilepro-cell/Private-Claude.git
cd Private-Claude/agent-4.0

# Install dependencies
pip install -r ../requirements.txt

# Optional: Install PDF tools
pip install PyPDF2 pdfplumber reportlab
```

### Launch

#### Option 1: Auto-detect Role
```bash
python setup_and_launch.py
```

#### Option 2: Specify Role
```bash
# Non-Coder interface
python setup_and_launch.py --role non_coder

# Python Developer interface
python setup_and_launch.py --role python_developer

# AI Developer interface
python setup_and_launch.py --role ai_developer

# CFO Deep Sync interface
python setup_and_launch.py --role cfo_deep_sync
```

#### Option 3: Specify Device
```bash
# Phone interface
python setup_and_launch.py --role non_coder --device phone

# Tablet interface
python setup_and_launch.py --role non_coder --device tablet
```

### Check Dependencies Only
```bash
python setup_and_launch.py --check-only
```

### Setup Without Launching
```bash
python setup_and_launch.py --setup-only
```

## Troubleshooting

### Dependencies Not Found
```bash
pip install -r requirements.txt
pip install PyPDF2 pdfplumber reportlab  # Optional
```

### Configuration Not Loaded
Check that `config/agent_versions_config.json` exists and is valid JSON.

### Interface Not Launching
Ensure Python 3.8+ is installed:
```bash
python --version
```

### Remote Access Not Working
Check firewall settings and ensure port 8080 is open:
```bash
sudo ufw allow 8080
```
